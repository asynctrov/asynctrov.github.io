Oracle数据库常用语句

查询所登录用户的sid

select instance_name from v$instance;

查询当前登录用户所在的表空间

select username,default_tablespace from dba_users where username='登录的用户名(大写)'

查询所有表空间

Select a.Tablespace_Name,
       Round(Sum(Dbsize) / 1024 / 1024 / 1024, 2) "dbszie(G)",
       Round(Sum(Fsize) / 1024 / 1024 / 1024, 2) "fszie(G)",
       Round((Sum(Dbsize) - Sum(Fsize)) / Sum(Dbsize), 4) * 100 "利用率",
       Round(Sum(Fsize) / Sum(Dbsize), 4) * 100 "空闲率"
  From (Select Tablespace_Name, Sum(Bytes) Dbsize
          From Dba_Data_Files
         Group By Tablespace_Name) a,
       (Select Tablespace_Name, Sum(Bytes) Fsize
          From Dba_Free_Space
         Group By Tablespace_Name) b
 Where a.Tablespace_Name = b.Tablespace_Name(+)
 Group By a.Tablespace_Name;
创建sequence
create sequence YBAQ02_SEQNO

查看导致锁表的sql语句是那一条

select l.session_id sid, 
s.serial#, 
l.locked_mode, 
l.oracle_username, 
s.user#, 
l.os_user_name, 
s.machine, 
s.terminal, 
a.sql_text, 
a.action 
from v$sqlarea a, v$session s, v$locked_object l 
where l.session_id = s.sid 
and s.prev_sql_addr = a.address 
order by sid, s.serial#;

查询表空间存放路径

select upper(dd.tablespace_name) "表空间名",dd.file_name "表空间路径", round(sum(dd.bytes)/(1024*1024),2) "表空间大小(M)"
        from sys.dba_data_files dd
        group by dd.tablespace_name,dd.file_name;

添加表空间

alter tablespace 表空间名(jxpyc) add datafile '/u01/app/oracle/oradata/hznet/res16.dbf' size 4096m;

查看表空间利用率

Select a.Tablespace_Name,
       Round(Sum(Dbsize) / 1024 / 1024 / 1024, 2) "dbszie(G)",
       Round(Sum(Fsize) / 1024 / 1024 / 1024, 2) "fszie(G)",
       Round((Sum(Dbsize) - Sum(Fsize)) / Sum(Dbsize), 4) * 100 "利用率",
       Round(Sum(Fsize) / Sum(Dbsize), 4) * 100 "空闲率"
  From (Select Tablespace_Name, Sum(Bytes) Dbsize
          From Dba_Data_Files
         Group By Tablespace_Name) a,
       (Select Tablespace_Name, Sum(Bytes) Fsize
          From Dba_Free_Space
         Group By Tablespace_Name) b
 Where a.Tablespace_Name = b.Tablespace_Name(+)
 Group By a.Tablespace_Name;

查询表空间

SELECT
	upper( dd.tablespace_name ) "表空间名",
	dd.file_name "表空间路径",
	round( sum( dd.bytes ) / ( 1024 * 1024 ), 2 ) "表空间大小(M)" 
FROM
	sys.dba_data_files dd 
GROUP BY
	dd.tablespace_name,
	dd.file_name;

查询表空间

select
       b.file_name 物理文件名,
       b.tablespace_name 表空间,
       b.bytes/1024/1024大小M,
      (b.bytes-sum(nvl(a.bytes,0)))/1024/1024  已使用M,
      substr((b.bytes-sum(nvl(a.bytes,0)))/(b.bytes)*100,1,5)  利用率
from dba_free_space a,dba_data_files b
where a.file_id=b.file_id 
group by b.tablespace_name,b.file_name,b.bytes
order by b.tablespace_name;
